Object.values

    
const sales = {
    John: 30,
    Sally: 40,
    Rob: 35,
    Jim: 50,
    Kim: 49
};

function sumSales(obj){
    let sum =0;
    for(let value of Object.values(obj)){
        sum += value;
    }
    return sum;
}

console.log(sumSales(sales));
/* output: 
    204
*/
